\begin{frame}{Thuật toán Cocks-Pinch truyền thống}
  \centering
  \begin{tikzpicture}[
    node distance=0.5cm and 0.5cm,
    compute/.style={
      draw=clrPurple600, fill=clrPurple50, text=clrPurple800,
      rounded corners=3pt, align=center,
      minimum width=1.5cm, minimum height=0.8cm,
      inner sep=3pt, font=\tiny\sffamily
    },
    sample/.style={
      draw=clrBlue600, fill=clrBlue50, text=clrBlue800,
      rounded corners=3pt, align=center,
      minimum width=1.5cm, minimum height=0.8cm,
      inner sep=3pt, font=\tiny\sffamily
    },
    decision/.style={
      diamond, aspect=1.7,
      draw=clrAmber600, fill=clrAmber50, text=clrAmber800,
      align=center, font=\tiny\sffamily, inner sep=1pt
    },
    io/.style={
      draw=clrTeal600, fill=clrTeal50, text=clrTeal800,
      rounded corners=3pt, align=center,
      minimum width=1.5cm, minimum height=0.8cm,
      inner sep=3pt, font=\tiny\sffamily
    },
    input/.style={
      draw=clrGray600, fill=clrGray50, text=clrGray800,
      rounded corners=3pt, align=center,
      minimum width=1.5cm, minimum height=0.8cm,
      inner sep=3pt, font=\tiny\sffamily
    },
    arr/.style={-{Stealth[length=4pt]}, semithick, draw=clrGray600},
    yes/.style={font=\tiny\sffamily, text=clrGreen600},
    no/.style={font=\tiny\sffamily, text=clrAmber800},
    lbl/.style={font=\tiny\itshape\sffamily, text=clrGray600},
  ]
    %% Row 1: Input -> r -> checkr -> t0 -> y0
    \node[input] (start) {$k,\; D<0$\\$T \in [T_0, T_{\max}]$\\$\lambda_r,\; \lambda_p$};
    \node[compute, right=of start] (compr) {$r = \Phi_k(T)$};
    \node[decision, right=0.6cm of compr] (checkr)
      {$r$ nguyên tố\\$|r|=\lambda_r$\\$\sqrt{-D}\!\pmod{r}$?};
    \node[compute, right=0.6cm of checkr] (compt0)
      {$t_0 = T^i+1 \pmod{r}$};
    \node[compute, right=of compt0] (compy0)
      {$y_0 = \frac{t_0-2}{\sqrt{-D}} \pmod{r}$};

    %% Row 1 arrows
    \draw[arr] (start)  -- (compr);
    \draw[arr] (compr)  -- (checkr);
    \draw[arr] (checkr) -- node[yes, above] {Có} (compt0);
    \draw[arr] (compt0) -- (compy0);

    %% checkr "Không" loop: up -> left -> down to compr
    \draw[arr] (checkr.north) -- ++(0,0.6)
      node[no, right, pos=0.5] {Không} -| (compr.north);

    %% Labels row 1
    \node[lbl, below=0.3cm of compr] {$T = T_0 \ldots T_{\max}$};
    \node[lbl, above=0.2cm of compt0] {$i:\;\gcd(i,k)=1$};

    %% Row 2: y0 -> sample -> p -> checkp -> output
    \node[sample, below=1.3cm of compy0] (sample)
      {$t = t_0 + h_t r$\\$y = y_0 + h_y r$};
    \node[compute, left=of sample] (compp)
      {$p = \frac{t^2+Dy^2}{4}$};
    \node[decision, left=0.6cm of compp] (checkp)
      {$p\!\equiv\!1\!\pmod{k}$\\$p$ nguyên tố?};
    \node[io, left=0.6cm of checkp] (output)
      {$(p,r,T,t,y)$};

    %% Connector: compy0 -> sample
    \draw[arr] (compy0.south) -- ++(0,-0.4) -| (sample.north);

    %% Row 2 arrows (right to left)
    \draw[arr] (sample) -- (compp);
    \draw[arr] (compp)  -- (checkp);
    \draw[arr] (checkp) -- node[yes, above] {Có} (output);

    %% checkp "Không" loop back to sample
    \draw[arr] (checkp.south) -- ++(0,-0.6)
      node[no, right, pos=0.5] {Không} -| (sample.south);

    %% Label row 2
    \node[lbl, below=0.8cm of sample] {thử lại $h_t,\,h_y$};
  \end{tikzpicture}
\end{frame}
