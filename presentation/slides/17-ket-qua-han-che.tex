\begin{frame}[label=sec4]{Kết quả đạt được \& Hạn chế}
  \begin{columns}[t]
    \column{0.5\textwidth}
      \begin{block}{Kết quả}
        \begin{itemize}
          \item[$\checkmark$] Sinh đường cong KSS18 1024-bit\\
                với cả $\mathbb{F}_r$, $\mathbb{F}_p$ hỗ trợ NTT
          \item[$\checkmark$] Bảo mật 256-bit đối xứng\\
                (Pollard-$\rho$: $O(2^{256})$, MOV: 18432-bit)
          \item[$\checkmark$] Miễn nhiễm Anomalous ($\#E \neq p$)\\
                và Invalid Curve (kiểm tra khởi tạo)
          \item[$\checkmark$] Schnorr \& ECDSA hoạt động đúng
          \item[$\checkmark$] Không hằng số ``nothing-up-my-sleeve''
        \end{itemize}
      \end{block}
    \column{0.5\textwidth}
      \begin{alertblock}{Hạn chế}
        \begin{itemize}
          \item \textbf{Timing side-channel}\\
                Double-and-Add rò rỉ bit scalar\\
                $\Rightarrow$ cần Montgomery Ladder
          \item \textbf{Tọa độ Affine}\\
                1 nghịch đảo/phép cộng điểm\\
                $\Rightarrow$ cần Jacobian/Projective
          \item \textbf{Hiệu năng}\\
                $\sim$8000$\times$ chậm hơn secp256k1\\
                (trường lớn $4\times$ + chưa tối ưu ASM)
        \end{itemize}
      \end{alertblock}
  \end{columns}
\end{frame}
