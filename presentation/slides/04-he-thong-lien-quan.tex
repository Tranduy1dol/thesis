\begin{frame}{Các hệ thống liên quan}
  \begin{table}
    \small
    \begin{tabular}{lcccc}
      \toprule
      Đường cong & $|p|$ & Bảo mật & NTT ($\nu_2$) & Minh bạch \\
      \midrule
      secp256k1 & 256 bit & 128-bit & Không & Có \\
      P-256 (NIST) & 256 bit & 128-bit & Không & \textcolor{red!70!black}{\textbf{Không}} \\
      BLS12-381 & 381 bit & 128-bit & $\nu_2(r{-}1) = 32$ & Có \\
      \bottomrule
    \end{tabular}
  \end{table}
  \vspace{0.3em}
  \textbf{Hạn chế của các hệ thống hiện tại:}
  \begin{itemize}
    \item Mức bảo mật cố định 128-bit --- không đủ cho kịch bản dài hạn
    \item Tham số không minh bạch (NIST P-256: seed không giải thích)
    \item BLS12-381: chỉ $\mathbb{F}_r$ hỗ trợ NTT, $\mathbb{F}_p$ thì không
  \end{itemize}
