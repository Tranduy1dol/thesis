\begin{frame}[label=sec1]{Bối cảnh}
  \begin{block}{ECC là nền tảng bảo mật hạ tầng số}
    TLS 1.3, SSH, Bitcoin (secp256k1), Ethereum (BLS12-381), chữ ký số điện tử
  \end{block}
  \vspace{0.4em}
  \textbf{Thực trạng:}
  \begin{itemize}
    \item Hầu hết hệ thống \textbf{tái sử dụng} đường cong sinh sẵn bởi NIST, SECG
    \item Tham số dùng ``hạt giống'' không giải thích được $\Rightarrow$ nguy cơ backdoor
    \item Sự kiện \texttt{Dual\_EC\_DRBG}: NSA bị cáo buộc cài cửa hậu vào PRNG chuẩn NIST
    \item Mức bảo mật phổ biến 128-bit --- chưa đáp ứng yêu cầu dài hạn NIST 2030+
  \end{itemize}
\end{frame}
