\begin{frame}{Thuật toán Cocks-Pinch cải tiến}
  \textbf{Điều kiện CM:} $4p = t^2 + 3v^2$,\quad
  $r = \Phi_{18}(T) = T^6 - T^3 + 1$
  \vspace{0.5em}

  \begin{columns}[t]
    \column{0.5\textwidth}
      \begin{alertblock}{Cải tiến 1: NTT-friendly $r$}
        Nhận xét: $r - 1 = T^3(T^3 - 1)$\\[4pt]
        Nếu $T \equiv 0 \pmod{2^{11}}$ thì\\
        $v_2(r-1) = 3\times 11 = 33 \geq 32$\\[4pt]
        $\Rightarrow$ \textbf{Không cần rejection sampling}
      \end{alertblock}
    \column{0.5\textwidth}
      \begin{alertblock}{Cải tiến 2: NTT-friendly $p$}
        Mở rộng lưới $(h_t, h_y)$ trong bước nâng:\\
        $p = \dfrac{(t_0 + h_t r)^2 + 3(y_0 + h_y r)^2}{4}$\\[4pt]
        $\Rightarrow$ kiểm soát two-adicity của $p-1$
      \end{alertblock}
  \end{columns}
  \vspace{0.5em}

      Thời gian thực tế & 2--10 giây & 30--60 giây \\

\end{frame}
