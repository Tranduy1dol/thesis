        \begin{enumerate}
          \item \textbf{Tự chủ mật mã}\\
                Sinh đường cong bằng quy trình minh bạch, không seed bí ẩn
          \item \textbf{Trần bảo mật}\\
                ECC 256-bit $\to$ 128-bit\\
                Cần 1024-bit $\to$ \textbf{256-bit}
          \item \textbf{An toàn bộ nhớ}\\
                Loại bỏ lớp lỗi buffer overflow bằng ngôn ngữ Rust
        \end{enumerate}
      \end{alertblock}
    \column{0.5\textwidth}
      \begin{block}{Yêu cầu kỹ thuật}
        \begin{itemize}
          \item Đường cong pairing-friendly\\($k = 18$, hỗ trợ BLS)
          \item Cấu trúc NTT-friendly\\($\nu_2(p-1), \nu_2(r-1) \geq 32$)
          \item Tham số có thể kiểm chứng\\(từ seed $T$ công khai)
          \item Chữ ký số hoạt động\\(Schnorr, ECDSA)
        \end{itemize}
      \end{block}
