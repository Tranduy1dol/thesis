\begin{frame}{Phát biểu bài toán}
  \begin{columns}[t]
    \column{0.5\textwidth}
      \begin{alertblock}{Vấn đề cần giải quyết}
        \begin{enumerate}
          \item \textbf{Tự chủ mật mã}\\
                Sinh đường cong bằng quy trình minh bạch, không seed bí ẩn
          \item \textbf{Trần bảo mật}\\
                ECC 256-bit $\to$ 128-bit\\
                Cần 1024-bit $\to$ \textbf{256-bit}
        \end{enumerate}
      \end{alertblock}
    \column{0.5\textwidth}
      \begin{block}{Thách thức}
        \begin{itemize}
          \item Sinh đường cong 1024-bit\\tốn kém tính toán
          \item Phải đảm bảo kháng\\các tấn công hiện đại (TNFS)
          \item Cần chứng minh tính đúng đắn\\toán học cho mọi phép toán
          \item Không dựa vào thư viện\\mật mã có sẵn
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}
